\begin{frame}{Compare $\langle \lambda_1 \rangle$ with $\tau_{Kolmogorov}^{-1}$}
  \begin{equation}
    \langle \lambda_1 \rangle = \langle \tr (S^2) \rangle
  \end{equation}
  \begin{equation}
    \tau_{Kolmogorov}^{-1} = \left\langle \norm{S^2} \right\rangle = \left\langle \sqrt{\tr(S^2)} \right\rangle
  \end{equation}
\end{frame}
